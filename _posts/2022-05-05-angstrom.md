from pwn import *
import string

alpha = string.ascii_lowercase
def decrypt(ct, key):
  ret = ""
  i = 0
  for c in ct:
    if c in alpha:
      ret += alpha[(alpha.index(c) - alpha.index(key[i])) % len(alpha)]
      i = (i + 1) % len(key)
    else:
      ret += c
  return ret

def solve(i):
  info(f"Solving Round {i}")
  io.recvuntil(f"Challenge {i}: ".encode())
  chall = io.recvline().decode().strip()
  
  for start in range(len(chall)):
    encrypted = chall[start:]
    key = decrypt(encrypted[:4], "actf")
    try:
      fleg = decrypt(encrypted, key).split("fleg")[0]
      if fleg.startswith("actf{") and fleg.endswith("}") \
          and fleg.count("{") == 1 and fleg.count("}") == 1:
        info(f"Flag Round {i}: {fleg}")
        io.sendlineafter(b"> ", fleg.encode())
        break
    except:
      pass

if __name__ == "__main__":
  # io = process(["python3", "main.py"])
  io = remote("challs.actf.co", 31333)

  for i in range(50):
    solve(i)

// Uint64 returns a non-negative pseudo-random 64-bit integer as a uint64.
func (rng *rngSource) Uint64() uint64 {
  rng.tap--
  if rng.tap < 0 {
    rng.tap += rngLen
  }

  rng.feed--
  if rng.feed < 0 {
    rng.feed += rngLen
  }

  rng.vec[rng.feed] = x
  return uint64(x)
}

ƒê√¢y l√† m·ªôt b√†i RSA, ta ƒë∆∞·ª£c bi·∫øt $$n, e, c$$ v√† th√™m gi√° tr·ªã "hint" $$f$$ th·ªèa m√£n:

$$
f = c^{d'} \pmod{n}
$$

v·ªõi $$d' = d \pm 2^{i} \pm 2^{j} \pm 2^{k}$$ v√† $$d = e^{-1} \pmod{\phi(n)}$$. N·∫øu ta t√¨m ƒë∆∞·ª£c b·ªô $$(i, j, k)$$ th√¨ d·ªÖ d√†ng t√¨m ra `flag` b·∫±ng c√°ch t√≠nh:

$$
m = f \times c^{\mp 2^{i} \mp 2^{j} \mp 2^{k}} \pmod{n}
$$

·ªû ƒë√¢y ta kh√¥ng th·ªÉ bruteforce ƒë∆∞·ª£c, $$\binom{4096}{3} \sim 2^{34}$$ kh√° l·ªõn, tuy nhi√™n $$\binom{4096}{2} \sim 2^{23}$$ th√¨ c√≥ th·ªÉ tr√¢u ƒë∆∞·ª£c! Do ƒë√≥ m·ªôt c√°ch t·ª± nhi√™n ta nghƒ© t·ªõi ph∆∞∆°ng ph√°p MITM ƒë·ªÉ gi·∫£m ƒë·ªô ph·ª©c t·∫°p.

$$
\begin{align*}
  f^{e} &= c^{ed'} = c^{e(d \pm 2^{i} \pm 2^{j} \pm 2^{k})} \pmod{n} \\
  \Rightarrow f^{e} \times \left(c^e\right)^{\mp 2^{k}} &= c \times \left(c^e\right)^{\pm 2^{i} \pm 2^{j}} \pmod{n}
\end{align*}
$$

K·∫øt h·ª£p v·ªõi multiprocessing th√¨ t·∫ßm $$30$$ ph√∫t l√† xongüòÑ
